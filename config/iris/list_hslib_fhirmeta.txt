zn "HSLIB"
write "=== FHIRMeta Storage classes in HSLIB ===",!
set stmt = ##class(%SQL.Statement).%New()
do stmt.%Prepare("SELECT Name FROM %Dictionary.ClassDefinition WHERE Name LIKE 'HS.FHIRMeta.Storage%' ORDER BY Name")
set rs = stmt.%Execute()
for {
    quit:'rs.%Next()
    write rs.Name,!
}
halt
