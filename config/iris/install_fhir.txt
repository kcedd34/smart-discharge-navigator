// FHIR Server Installation Script for InterSystems IRIS
// Based on InterSystems FHIR R4 documentation

// 1. Unexpire passwords (development only)
set $NAMESPACE = "%SYS"
write "Unexpiring user passwords...",!
do ##class(Security.Users).UnExpireUserPasswords("*")
write "Passwords unexpired",!,!

// 2. Create namespace for FHIR server
set $NAMESPACE = "HSLIB"
write "Creating namespace 'FHIRSERVER'...",!
if '##class(%SYS.Namespace).Exists("FHIRSERVER") {
    do ##class(HS.Util.Installer.Foundation).Install("FHIRSERVER")
    write "Namespace FHIRSERVER created",!
} else {
    write "Namespace FHIRSERVER already exists",!
}

set $NAMESPACE = "FHIRSERVER"

// 3. Install FHIR R4 server
write !,"Installing FHIR R4 server...",!
set fhirServerPath = "/fhir/r4"
set strategyClass = "HS.FHIRServer.Storage.Json.InteractionsStrategy"
set metadataPackage = "hl7.fhir.r4.core@4.0.1"

write "Namespace: ",fhirServerPath,!
write "Strategy: ",strategyClass,!
write "Metadata: ",metadataPackage,!,!

do ##class(HS.FHIRServer.Installer).InstallNamespace()
do ##class(HS.FHIRServer.Installer).InstallInstance(fhirServerPath, strategyClass, metadataPackage)
write "FHIR server installed at: ",fhirServerPath,!,!

// 4. Configure search limits
write "Configuring search limits...",!
set strategy = ##class(HS.FHIRServer.API.InteractionsStrategy).GetStrategyForEndpoint(fhirServerPath)
set configData = strategy.GetServiceConfigData()
set configData.DefaultSearchPageSize = 1000
set configData.MaxSearchPageSize = 10000
set configData.MaxSearchResults = 10000
do strategy.SaveServiceConfigData(configData)
write "Search limits configured:",!
write "  - DefaultSearchPageSize: 1000",!
write "  - MaxSearchPageSize: 10000",!
write "  - MaxSearchResults: 10000",!,!

write "========================================",!
write "FHIR Server Installation Complete!",!
write "========================================",!
write "Endpoint: http://localhost:52773/fhir/r4/metadata",!
write "Username: SuperUser",!
write "Password: SYS",!
write !

halt
