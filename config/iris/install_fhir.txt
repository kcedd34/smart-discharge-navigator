// FHIR Server Installation Script for InterSystems IRIS
// All statements are single-line — required for piped stdin execution.
// Multi-line block syntax (FOR { }, IF { }) causes <SYNTAX> in piped mode.

// 1. Unexpire passwords (development only)
set $NAMESPACE = "%SYS"
write "Step 1: Unexpiring user passwords...",!
do ##class(Security.Users).UnExpireUserPasswords("*")
write "  Done.",!,!

// 2. Create FHIRSERVER namespace (if it does not exist yet)
set $NAMESPACE = "HSLIB"
write "Step 2: Ensuring FHIRSERVER namespace exists...",!
set needsInstall = '##class(%SYS.Namespace).Exists("FHIRSERVER")
if needsInstall do ##class(HS.Util.Installer.Foundation).Install("FHIRSERVER")
if needsInstall write "  Foundation.Install called. Waiting 45s for background jobs...",!
if needsInstall hang 45
write "  FHIRSERVER namespace ready.",!,!

// 3. Install FHIR R4 server
set $NAMESPACE = "FHIRSERVER"
write "Step 3: Installing FHIR R4 server...",!
do ##class(HS.FHIRServer.Installer).InstallNamespace()
do ##class(HS.FHIRServer.Installer).InstallInstance("/fhir/r4", "HS.FHIRServer.Storage.Json.InteractionsStrategy", "hl7.fhir.r4.core@4.0.1")
write "  FHIR server installed at: /fhir/r4",!,!

// 4. Configure search limits
write "Step 4: Configuring search limits...",!
set strategy = ##class(HS.FHIRServer.API.InteractionsStrategy).GetStrategyForEndpoint("/fhir/r4")
set configData = strategy.GetServiceConfigData()
set configData.DefaultSearchPageSize = 1000
set configData.MaxSearchPageSize = 10000
set configData.MaxSearchResults = 10000
do strategy.SaveServiceConfigData(configData)
write "  DefaultSearchPageSize: 1000",!
write "  MaxSearchPageSize:     10000",!
write "  MaxSearchResults:      10000",!,!

write "========================================",!
write "FHIR Server Installation Complete!",!
write "========================================",!
write "Endpoint: http://localhost:52773/fhir/r4/metadata",!
write "Username: SuperUser / Password: SYS",!
write !

halt
