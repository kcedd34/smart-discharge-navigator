set app = ##class(Security.Applications).%OpenId("/fhir/r4")
write "App found: ",$isobject(app),!
write "Name: ",app.Name,!
write "AutheEnabled: ",app.AutheEnabled,!
write "MatchRoles: "
set tList = app.MatchRoles
set tKey = ""
for {
    set tKey = tList.Next(tKey)
    quit:tKey=""
    write tList.GetAt(tKey)," ; "
}
write !
write "ServicePath: ",app.ServicePath,!
write "DispatchClass: ",app.DispatchClass,!
halt
